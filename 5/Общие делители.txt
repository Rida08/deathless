#include <iostream>
#include <string>
#include <algorithm>
#include <vector>

using namespace std;
using ll = long long;

ll gcd(ll a, ll b) {
	if (!b) {
		return a;
	}
	return gcd(b, a % b);
}

int main(){
	int n;
	cin >> n;
	vector<ll>a(n);
	cin >> a[0];
	ll g = a[0];
	for (int i = 1; i < n; ++i) {
		cin >> a[i];
		g = gcd(a[i], g);
	}
	int s = 0;
	for (ll i = 1; i*i < g; ++i) {
		if (g % i == 0) {
			s++;
			if (i * i != g) {
				s++;
			}
		}
	}
	if (g == 1) {
		s = 1;
	}
	cout <<s;
}

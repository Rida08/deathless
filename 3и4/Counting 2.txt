	#include <iostream>
	#include <string>
	#include <algorithm>
	#include <vector>
	using namespace std;
	int main()
	{
		long n, q;
		cin >> n >> q;
		vector<int>a(n,0);
		long* x = new long[q];
		for (int i = 0; i < n; ++i) cin >>a[i];
		sort(begin(a), end(a));
		string an = "";
		for (int i = 0; i < q; ++i) {
			cin >> x[i];
			long l = 0, r = n, m = r / 2;
			while (r - l != 1) {
				a[m] >= x[i] ? r = m : l = m;
				m = (l + r) / 2;
			}
			an += to_string(n - r) + "\n";
		}
		cout << an;
	}

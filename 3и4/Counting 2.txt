#include <iostream>
#include <string>
#include <algorithm>
#include <vector>

using namespace std;
using ll = long long;

ll n, q, x;
string an = "";
vector<ll>a;

void solve(){
	ll l = 0, r = n-1;
	while (r - l != 1) {
		ll m = (l + r) / 2;
		if (a[m] >= x){
			l = m;
		}
		else {
			r = m;
		}
	}
	if (r < n - 1 && a[r + 1] > x) {
		an += to_string(r+1) + "\n";
	}else {
		an += to_string(r) + "\n";
	}
}

int main(){
	cin >> n >> q;
	ll c;
	for (int i = 0; i < n; ++i) {
		cin >> c;
		a.push_back(c);
	}
	sort(rbegin(a), rend(a));
	for (int i = 0; i < q; ++i) {
		cin >> x;
		if (a[0] < x) {
			an += "0\n";
		}
		else if (a[n-1]>=x) {
			an += to_string(n) + "\n";
		}else {
			solve();
		}
	}
	cout << an;
}

#include <iostream>
#include <string>
#include <algorithm>
#include <vector>
using ll = long long;
using namespace std;
int main()
{
	ll t, n, q, l, x;
	cin >> t;
	string an = "";
	for (ll i = 0; i < t; ++i) {
		cin >> n >> q;
		vector<ll>a(n, 0);
		vector<ll>s(n, 0);
		for (ll j = 0; j < n; ++j) cin >> a[j];
		sort(a.begin(), a.end());
		s[0] = a[0];
		for (ll j = 1; j < n; ++j)	s[j]	= s[j - 1] + a[j];
		for (ll j = 0; j < q; ++j) {
			cin >> x;
			if (s[n - 1] < x) an += "-1\n";
			else {
				ll r = n, l = 0, ii=0;
				while (/*l + 1 < r*/ii<60) {
					ll m = (l + r) / 2;
					s[m] > x ? r = m : l = m;
					ii++;
				}
				an += to_string(l-1) + "\n";
			}
		}
	}
	cout << an;
}

#include <iostream>
#include <string>
#include <algorithm>
#include <vector>

using ll = long long;
using namespace std;
int main(){
	int n, k, l;
	cin >> n >> k;
	vector<ll> s(n, 0);
	for (int i = 0; i < n; ++i) {
		for (int j = 0; j < 3; ++j) {
			cin >> l;
			s[i] += l;
		}
	}
	for (int i = 0; i < n; ++i) {
		vector<ll> s1;		
		for (int j = 0; j < n; j++) {
			s1.push_back(s[j]);
		}
		s1[i] += 300;
		int a = s1[i];
		sort(s1.rbegin(), s1.rend());
		if (s1[k-1]<=a)
			cout << "Yes\n";
		else
			cout << "No\n";
	}
}

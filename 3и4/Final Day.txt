#include <iostream>
#include <string>
#include <algorithm>
#include <vector>
using namespace std;
int main()
{
	int n, k,l;
	cin >> n >> k;
	vector<int> s(n, 0);
	for (int i = 0; i < n; ++i) {
		for (int j = 0; j < 3; ++j) {
			cin >> l;
			s[i] += l;
		}
	}
	for (int i = 0; i < n; ++i) {
		vector<int> s1;
		copy(s.begin(), s.end(), back_inserter(s1));
		s1[i] += 300;
		int a = s1[i];
		sort(begin(s1), end(s1));
		s1.resize(k);
		auto p = find(s1.begin(), s1.end(), a);
		if (p == s1.end())
			cout << "YES\n";
		else
			cout << "NO\n";
	}
}
